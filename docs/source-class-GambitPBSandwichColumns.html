 75:         $columnStyles = '';

 78:         // 移除没用的 jQuery sortable classes
 79:         $html = preg_replace( '/(ui-sortable-handle|ui-sortable)/', '', $content );

 82:         $tables = $html->find( 'table.pbsandwich_column' );

 94:             // 移除没有任何效果的编辑器样式

 97:             $tableStyles = trim( preg_replace( '/(^|\s)width:[^;]+;?\s?/', '', $tableStyles ) );
 98:             $tableStyles = trim( preg_replace( '/(^|\s)height:[^;]+;?\s?/', '', $tableStyles ) );

102:             if( !empty( $tableStyles ) ) {
103:                 $columnStyles .= '.sandwich.pbsandwich_column_%' . ( count( $hashes ) + 1 ) . '$s { ' . wp_kses( $tableStyles, [ ], [ ] ) . ' }';
104:             }
105:             $styleDump .= esc_attr( $tableStyles );

107:             foreach ( $tr->children() as $key => $td ) {
108:                 if( $td->tag != 'td' ) {

116:                 if( preg_match( '/^<(a|abbr|acronym|b|bdo|big|br|cite|code|dfn|em|i|img|input|kbd|label|q|samp|select|small|span|strong|sub|sup|textarea|tt|var|button|del|ins|map|object|script)[^>]+>/', $innerHTML ) === 1 ) {
117:                     $innerHTML = '<p>' . $innerHTML . '</p>';

119:                     $innerHTML = '<p>' . $innerHTML . '</p>';

122:                 // 移除空白自负Remove blank classes
123:                 $innerHTML = preg_replace( '/\sclass=[\'"]\s*[\'"]/', '', $innerHTML );
124: 
125:                 // 清理结尾 Cleanup ends
126:                 $innerHTML = trim( $innerHTML );
127: 
128:                 // 移除宽度, 使用 class 代替Remove the widths since we are using classes for those:
129:                 $columnStyle = trim( preg_replace( '/(^|\s)width:[^;]+;\s?/', '', $td->style ) );
130: 
131:                 // 获取分栏样式, 因为暂时没有 唯一 ID, 使用占位符作为 ID
132:                 if( !empty( $columnStyle ) ) {
133:                     $columnStyles .= '.sandwich.pbsandwich_column_%' . ( count( $hashes ) + 1 ) . '$s > div > div:nth-of-type(' . ( $key + 1 ) . ') { ' . wp_kses( $columnStyle, [ ], [ ] ) . ' }';
134:                 }
135:                 $styleDump .= esc_attr( $td->style );
136: 
137:                 // 获取所有分栏数据属性
138:                 $dataAttributes = '';
139:                 foreach ( $td->getAllAttributes() as $key => $value ) {

153:             $hash = crc32( $styleDump );

162:             $tableClasses[] = $html->find( 'table.pbsandwich_column', 0 )->class;

169:             $dataAttributes = '';

188:             $tables = $html->find( 'table.pbsandwich_column' );
189:         }
190: 
191:         // Sanitize the output for security
192:         $columnStyles = wp_kses( $columnStyles, [ ], [ ] );
193:         // Make sure our html entities are correct to make our rules work properly
194:         $columnStyles = html_entity_decode( $columnStyles );
195: 
196:         // 插入#号
197:         foreach ( $hashes as $key => $hash ) {
198:             $columnStyles = str_replace( '%' . ( $key + 1 ) . '$s', $hash, $columnStyles );
199:         }
200: 
201:         // 移除空行
202:         $newHTML = (string)$html;
203: 
204:         // Remove weirdly appended paragraph tags (they are there for some odd reason)
205:         $newHTML = preg_replace( '/(<p[^>]*>)([^<]*$)/', '$2', $newHTML );
206: 
207:         // Sometimes, our rows/columns get tangled up inside paragraphs, get rid of those
208:         $newHTML = preg_replace( '/<p[^>]*>\s*(<div)/', '$1', $newHTML );
209:         $newHTML = preg_replace( '/(<\/div>)\s*<\/p>/', '$1', $newHTML );
210: 
211:         // Since we're dealing with wrong paragraph tags, remove other wrong stuff such as
212:         // multiple end paragraphs & multiple start paragraphs
213:         $newHTML = preg_replace( '/(<\/p>)\s*<\/p>/', '$1', $newHTML );
214:         $newHTML = preg_replace( '/<p[^>]+>\s*(<p[^>]+>)/', '$1', $newHTML );
215: 
216:         return [
217:             'content' => $newHTML,
218:             'styles'  => $columnStyles,
